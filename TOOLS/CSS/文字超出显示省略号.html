    function creatDotDotDot($p, lineNum) {
	lineNum = lineNum || 2;
	var pLineHeight = parseInt($p.css('line-height'));
	//根据行数设置wrap高度
	var wrapHeight = lineNum * pLineHeight;
	//精确筛选
	var $imgs = $p.find('img');
	if ($imgs.length === 0) {
		while ($p.outerHeight() > wrapHeight) {
			$p.text($p.text().replace(/(\w{1}|\W{1})(\.{3})?$/, '...'));
		}
	}
}
